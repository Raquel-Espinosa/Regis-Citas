Apps Script — API REST para Sheets
// Google Apps Script — Pegar en Extensions > Apps Script
const SHEET_ID = 'TU_SPREADSHEET_ID';

function doGet(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(e.parameter.tab || 'Citas');
  const data = sheet.getDataRange().getValues();
  const headers = data.shift();
  const rows = data.map(r => Object.fromEntries(headers.map((h,i) => [h,r[i]])));
  return ContentService.createTextOutput(JSON.stringify(rows))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const body = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(body.tab);
  sheet.appendRow(body.row);
  return ContentService.createTextOutput(JSON.stringify({ok:true}));
}
